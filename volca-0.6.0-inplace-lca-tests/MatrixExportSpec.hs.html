   45                 let offDiagonalLines = tail lines  -- Skip header
   46                 let offDiagonalEntries = filter (\l -> not (T.isInfixOf ";1.0;" l)) offDiagonalLines

   48                 -- For SAMPLE.min3: Expected -0.6 and -0.4
