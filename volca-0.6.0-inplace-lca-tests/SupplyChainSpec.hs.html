   77         let dummyFlow = FlowInfo { fiId = nil, fiName = "", fiCategory = "" }
   78             dummyEdge f t = TreeEdge { teFrom = f, teTo = t, teFlow = dummyFlow
   79                                      , teQuantity = 1.0, teUnit = "kg"
   80                                      , teEdgeType = TechnosphereEdge }
   81             makeNode nid name parent = ExportNode
   82                 { enId = nid, enName = name, enDescription = [], enLocation = ""
   83                 , enUnit = "", enNodeType = ActivityNode, enDepth = 0
   84                 , enLoopTarget = Nothing, enParentId = parent
   85                 , enChildrenCount = 0, enCompartment = Nothing }
   86             rootNode   = makeNode "r" "root activity"   Nothing
   87             midNode    = makeNode "m" "middle activity" (Just "r")
   88             leafNode   = makeNode "l" "leaf activity"   (Just "m")
   89             meta = TreeMetadata { tmRootId = "r", tmMaxDepth = 3, tmTotalNodes = 3
   90                                 , tmLoopNodes = 0, tmLeafNodes = 1, tmExpandableNodes = 0 }
   91             tree = TreeExport
   92                 { teTree = meta
   93                 , teNodes = M.fromList [("r", rootNode), ("m", midNode), ("l", leafNode)]
   94                 , teEdges = [dummyEdge "r" "m", dummyEdge "m" "l"]
   95                 }
   96 
   97         it "filter matching leaf keeps leaf + all ancestors" $ do
   98             let result = filterTreeExport "leaf" tree
   99             M.keys (teNodes result) `shouldMatchList` ["r", "m", "l"]
  100             length (teEdges result) `shouldBe` 2
  101 
  102         it "filter matching middle node keeps middle + root only" $ do
  103             let result = filterTreeExport "middle" tree
  104             M.keys (teNodes result) `shouldMatchList` ["r", "m"]

  106 
  107         it "filter matching root keeps root only" $ do
  108             let result = filterTreeExport "root" tree

  110             length (teEdges result) `shouldBe` 0
  111 
  112         it "filter with no match returns empty" $ do

  115             length (teEdges result) `shouldBe` 0
  116
