   74 -- entered oncespec = do
   75     describe "ILCD Process Parser" $ do
   76         it "parses classification from classificationInformation" $
   77             M.lookup "ILCDCategories" (classOf ilcdProcessWithClassification)
   78                 `shouldBe` Just "End-of-life treatment/Material recycling"
   79 
   80         it "parses multiple classification systems" $ do
   81             let cls = classOf ilcdProcessWithTwoClassifications
   82             M.lookup "ILCDCategories" cls `shouldBe` Just "Energy/Electricity"
   83             M.lookup "EcoSpold" cls `shouldBe` Just "Supply"
   84 
   85         it "produces empty classifications when none present" $
   86             classOf ilcdProcessNoClassification `shouldBe` M.empty
   87 
   88     -- -------------------------------------------------------------------
   89     -- buildSupplierIndex: UUID-keyed, no name indirection
   90     -- -------------------------------------------------------------------
   91     describe "buildSupplierIndex" $ do
   92 
   93         it "indexes reference exchanges by flow UUID" $ do
   94             let activities = M.fromList
   95                     [ ((actUUID1, prodUUID1), activityWithRefExchange flowUUID1)
   96                     , ((actUUID2, prodUUID2), activityWithRefExchange flowUUID2)
   97                     ]
   98                 idx = buildSupplierIndex activities
   99             M.lookup flowUUID1 idx `shouldBe` Just (actUUID1, prodUUID1)
  100             M.lookup flowUUID2 idx `shouldBe` Just (actUUID2, prodUUID2)
  101 
  102         it "does not index non-reference exchanges" $ do
  103             let activities = M.fromList
  104                     [ ((actUUID1, prodUUID1), activityWithInputExchange flowUUID1) ]
  105                 idx = buildSupplierIndex activities
  106             M.lookup flowUUID1 idx `shouldBe` Nothing
  107 
  108         it "two activities with same flow name but different UUIDs are both indexed" $ do
  109             -- This is the bug the old name-based code had: M.fromList on names
  110             -- would silently discard one. UUID keys have no such collision.
  111             let activities = M.fromList
  112                     [ ((actUUID1, prodUUID1), activityWithRefExchange flowUUID1)
  113                     , ((actUUID2, prodUUID2), activityWithRefExchange flowUUID2)
  114                     ]
  115                 idx = buildSupplierIndex activities
  116             M.size idx `shouldBe` 2
  117 
  118     -- -------------------------------------------------------------------
  119     -- fixActivityExchanges: resolves input exchanges via supplier index
  120     -- -------------------------------------------------------------------
  121     describe "fixActivityExchanges" $ do
  122 
  123         it "resolves input exchange flow UUID to supplier (actUUID, prodUUID)" $ do
  124             let idx = M.fromList [(flowUUID1, (actUUID1, prodUUID1))]
  125                 act = activityWithInputExchange flowUUID1
  126                 fixed = fixActivityExchanges idx act
  127             case exchanges fixed of
  128                 [TechnosphereExchange { techFlowId = fid, techActivityLinkId = alink }] -> do
  129                     fid   `shouldBe` prodUUID1
  130                     alink `shouldBe` actUUID1
  131                 _ -> expectationFailure "expected one TechnosphereExchange"
  132 
  133         it "leaves input exchange unchanged when flow UUID not in index" $ do
  134             let idx = M.empty
  135                 act = activityWithInputExchange flowUUID1
  136                 fixed = fixActivityExchanges idx act
  137             case exchanges fixed of
  138                 [TechnosphereExchange { techFlowId = fid }] ->
  139                     fid `shouldBe` flowUUID1
  140                 _ -> expectationFailure "expected one TechnosphereExchange"
  141 
  142         it "does not touch output (reference) exchanges" $ do
  143             let idx = M.fromList [(flowUUID1, (actUUID1, prodUUID1))]
  144                 act = activityWithRefExchange flowUUID1
  145                 fixed = fixActivityExchanges idx act
  146             case exchanges fixed of
  147                 [TechnosphereExchange { techFlowId = fid, techIsReference = isRef }] -> do
  148                     fid   `shouldBe` flowUUID1   -- unchanged
  149                     isRef `shouldBe` True
  150                 _ -> expectationFailure "expected one TechnosphereExchange"
