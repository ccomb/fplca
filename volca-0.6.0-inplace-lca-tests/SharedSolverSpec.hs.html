   62 min3Solver :: IO (SharedSolver, Database)
   63 -- entered oncemin3Solver = do
   64     db <- loadSampleDatabase "SAMPLE.min3"
   65     let techTriples = [ (fromIntegral i, fromIntegral j, v)
