   14 -- entered oncespec = do
   15     describe "computeProcessLCIAContributions - SAMPLE.min3" $ do
   16         it "only Z has a non-zero contribution for a CF=1 CO2 method" $ do
   17             db <- loadSampleDatabase "SAMPLE.min3"
   18             let co2Flow = findFlowByName db "carbon dioxide"
   19             case co2Flow of
   20                 Nothing  -> expectationFailure "CO2 flow not found in SAMPLE.min3"
   21                 Just co2 -> do
   22                     scalingVec <- computeScalingVector db 0
   23                     let cfMap = M.singleton (flowId co2) 1.0
   24                     let contribs = computeProcessLCIAContributions db scalingVec cfMap
   25                     let nonZero = M.filter (\v -> abs v > 1e-15) contribs
   26                     -- In SAMPLE.min3 only Z emits CO2 directly
   27                     M.size nonZero `shouldBe` 1
   28 
   29         it "total contribution equals sampleMin3ExpectedCO2 (0.96 kg)" $ do
   30             db <- loadSampleDatabase "SAMPLE.min3"
   31             let co2Flow = findFlowByName db "carbon dioxide"
   32             case co2Flow of
   33                 Nothing  -> expectationFailure "CO2 flow not found in SAMPLE.min3"
   34                 Just co2 -> do
   35                     scalingVec <- computeScalingVector db 0
   36                     let cfMap = M.singleton (flowId co2) 1.0
   37                     let contribs = computeProcessLCIAContributions db scalingVec cfMap
   38                     let total = sum (M.elems contribs)
   39                     withinTolerance defaultTolerance sampleMin3ExpectedCO2 total
   40                         `shouldBe` True
   41

   43             db <- loadSampleDatabase "SAMPLE.min3"

   52                             $ computeProcessLCIAContributions db scalingVec cfMap
